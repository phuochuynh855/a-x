<script>
    // CẤU HÌNH THÔNG TIN CỦA BẠN VÀO ĐÂY
    const firebaseConfig = {
        databaseURL: "https://mppt855-default-rtdb.asia-southeast1.firebasedatabase.app/"
    };
    
    firebase.initializeApp(firebaseConfig);
    const db = firebase.database();

    function connectFirebase() {
        const phone = document.getElementById('phoneInput').value;
        if(!phone) { alert("Vui lòng nhập số điện thoại"); return; }

        // Lấy dữ liệu thời gian thực
        db.ref('devices/' + phone).on('value', (snapshot) => {
            const data = snapshot.val();
            if(data) {
                document.getElementById('monitor').style.display = 'grid';
                document.getElementById('v_tong').innerText = data.voltage || "0.0";
                document.getElementById('a_sac').innerText = data.current || "0.0";
                document.getElementById('watt').innerText = (data.voltage * data.current).toFixed(1);
            } else {
                alert("Không tìm thấy thiết bị cho số điện thoại này!");
            }
        });
    }
</script>
